## 单向链表
### 定义
> 双向链表也叫双链表，是链表的一种，它的每个数据结点中都有两个指针，分别指向直接后继和直接前驱。所以，从双向链表中的任意一个结点开始，都可以很方便地访问它的前驱结点和后继结点。一般我们都构造双向循环链表。
### 实现
```js
class List {
  head = null

  static create (key) {
    return new Node(key)
  }

  print () {
    const res = []
    let currentNode = this.head
    while (currentNode) {
      res.push(currentNode.key)
      currentNode = currentNode.next
    }
    console.log(res.toString())
  }

  insert (node, positionKey) {
    if (!this.head) {
      this.head = node
    } else {
      let lastNode = this.head
      while (positionKey ? lastNode.key !== positionKey : lastNode.next) {
        lastNode = lastNode.next
      }

      if (lastNode.next) {
        lastNode.next.prev = node
        node.next = lastNode.next
      }
      lastNode.next = node
      node.prev = lastNode
    }
  }

  find (key) {
    let currentNode = this.head
    while (currentNode) {
      if (currentNode.key === key) {
        return currentNode
      } else {
        currentNode = currentNode.next || null
      }
    }
    throw Error('can not find ' + key)
  }

  delete (key) {
    let node = this.find(key)
    if (!node) { return console.log(key + ' is not exist') }

    if (this.head.key === key) {
      this.head = this.head.next
      this.head.prev = null
    } else {
      if (node.next) {
        const prev = node.prev
        const next = node.next
        prev.next = next
        next.prev = prev
        node.prev = node.next = null
      } else {
        // 最末尾
        node.prev.next = null
        node.prev = null
      }
    }
  }
}

class Node {
  next = null
  prev = null
  key = null

  constructor (key) {
    this.key = key
    this.next = null
    this.prev = null
  }
}

```

### 自测
```js
const list1 = new List()

const node1 = List.create(1)
const node2 = List.create(2)
const node3 = List.create(3)
const node4 = List.create(4)

list1.insert(node1)
list1.insert(node3)
list1.insert(node2)
list1.insert(node4, 1)
console.log(list1.find(3)) // {prevKey: 4, nextKey: 2}
list1.print() // 1,4,3,2

list1.delete(2) //1,4,3
list1.print()
console.log(list1.find(3)) // {prevKey: 4, nextKey: null}
```
