## 节流函数

### 实现
```js
function debounce (cb, delay) {
  let timer = null
  
  return function (...arg) {
    if (timer) {
      clearTimeout(timer)
    }
    
    timer = setTimeout(() => {
            cb.apply(this, arg)
          }, delay)
  }
}
```

### usage
```js
function showTop () {
  const scrollTop = document.body.scrollTop || document.documentElement.scrollTop
　console.log('滚动条位置：' + scrollTop)
}

window.onscroll = debounce(showTop, 1000)
```
