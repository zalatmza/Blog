## 节流函数

### 实现
```js
function throttle (cb, delay) {
  let prev = 0
  return function (...arg) {
    const now = Date.now()
    const that = this
    if (now - prev > delay) {
      cb.apply(that, arg)
      prev = now
    }
  }
}
```

### usage
```js
const print = throttle(p, 2000)

function p (a) {
  console.log(a)
}

while(1) {
  print(23)
}
```
