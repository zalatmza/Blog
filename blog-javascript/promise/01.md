# 实现Promise

## 一、简单版实现
### 基本用法
```js
const promise = new Promise((resolve, reject) => {
  if (xxx) {
    resolve()
  } else {
    reject()
  }
})

promise.then(a => {},e => {})
```

### 实现
```js
class Promise {
  value = undefined
  status = 'pending'
  
  successCb = () => null
  errorCb = () => null
  
  constructor (execCb) {
    const resolve = (value) => {
      if (this.status !== 'pending') { return }
      this.status = 'resolved'
      this.value = value
      this.successCb(this.value)
    }
    
    const reject = (error) => {
      if (this.status !== 'pending') { return }
      this.status = 'rejected'
      this.errorCb(error)
    }
    
    try {
      execCb(resolve, reject)
    } catch (e) {
      reject(e)
    }
  }
  
  then (resolveCb, rejectCb) {
    this.successCb = resolveCb
    this.errorCb = rejectCb
  }
}
```

### 测试
```js
const promise = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve('resolve')
  }, 100)
  setTimeout(() => {
    reject('reject')
  }, 1000)
})

promise.then(a => {
  console.log(a)
}, e => {
  console.log(e)
})
```
